DECLARE

    VEMPNO NUMBER(4);
    VENAME VARCHAR2(10);
    
BEGIN
    
    VEMPNO := 7788;
    VENAME := 'SCOTT';
    DBMS_OUTPUT.PUT_LINE('사원번호 / 사원이름');
    DBMS_OUTPUT.PUT_LINE(VEMPNO || '/' || VENAME);

END;
/

DECLARE
    VDNAME DEPT.DNAME%TYPE;    
BEGIN
    SELECT DEPT.DNAME INTO VDNAME FROM DEPT WHERE DEPTNO = 10;   
    
    DBMS_OUTPUT.PUT_LINE(VDNAME);
    
END;
/

DECLARE

    VEMP_RECORD EMP%ROWTYPE;
BEGIN
    SELECT EMPNO,ENAME,JOB INTO VEMP_RECORD.EMPNO,VEMP_RECORD.ENAME,VEMP_RECORD.JOB FROM EMP WHERE ENAME = 'SCOTT';
    
    DBMS_OUTPUT.PUT_LINE('사원번호 = ' || VEMP_RECORD.EMPNO);

END;
/

DECLARE

    TOT NUMBER(2) := 0;
BEGIN
    
    FOR I IN 1..10 LOOP
        TOT := TOT + I;
    END LOOP;
    
    DBMS_OUTPUT.PUT_LINE('1~10까지 범위의 정수 합계 =' || TOT);

END;
